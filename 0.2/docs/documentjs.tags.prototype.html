@prototype

  • constructor
documentjs.tags.prototype  
  • source

Declares that @property and @function tags belong to the preceeding @function's prototype object.

@prototype 

/**
 * @function
 * Creates an Animal
 */
Animal = function(){ ... }
/** @prototype */
Animal.prototype = {
   /**
    * Eats another animal.
    */
    eat: function(animal){ ... }
}
